running_processes

Only handles documents that have a field running_ids. Allows efficient finding of which raspberry pis are currently running.

    function(doc) {
if (!doc.running_ids) return;
emit([doc.created_by, doc.timestamp, doc.ip], null);
doc.running_ids.forEach( function(o) {
  emit([doc.created_by, doc.timestamp, null], { _id : o });
});
}
